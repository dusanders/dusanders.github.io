using System;
using System.IO;
using System.Collections.Generic;

namespace LICDataImporter
{
    public class LifePolicyParser
    {
        private static int POLICYNUMBER_LENGTH = 30;
        private static int POLICYHOLDERID_LENGTH = 20;
        private static int POLICYHOLDERDOB_LENGTH = 8;
        private static int FATHERSAGEATDEATH_LENGTH = 5;
        private static int MOTHERSAGEATDEATH_LENGTH = 5;
        private static int CIGS_LENGTH = 5;
        private static int SMOKING_LENGTH = 5;
        private static int BLOODPRESSURE_LENGTH = 4;
        private static int AVGFAT_LENGTH = 5;
        private static int HEARTDISEASE_LENGTH =1;
        private static int CANCER_LENGTH = 1;
        private static int HOSPITALIZED_LENGTH = 1;
        private static int DANGEROUS_LENGTH = 255;
        private static int POLICYSTART_LENGTH = 8;
        private static int POLICYEND_LENGTH = 8;
        private static int AGENTID_LENGTH = 20;
        private static int PAYOFF_LENGTH = 10;
        private static int PREMIUM_LENGTH = 10;
        public static List<string> policyNumbers = new List<string>();
        public static List<string> holderIds = new List<string>();
        public static List<DateTime> holderDOBs = new List<DateTime>();
        public static List<double> fathersAges = new List<double>();
        public static List<double> mothersAges = new List<double>();
        public static List<int> cigsPerDays = new List<int>();
        public static List<int> smokingHistories = new List<int>();
        public static List<int> bloodPressures = new List<int>();
        public static List<int> avgFats = new List<int>();
        public static List<bool> heartDiseases = new List<bool>();
        public static List<bool> cancers = new List<bool>();
        public static List<bool> hospitalizeds = new List<bool>();
        public static List<string> dangerousActs = new List<string>();
        public static List<DateTime> policyStarts = new List<DateTime>();
        public static List<DateTime> policyEnds = new List<DateTime>();
        public static List<string> agentIds = new List<string>();
        public static List<double> payoffs = new List<double>();
        public static List<double> premiums = new List<double>();
        public LifePolicyParser(string fileName)
        {
            StreamReader reader = new StreamReader(fileName);
            string inputString = null;
            while((inputString = reader.ReadLine()) != null)
            {
                char[] inputArray = inputString.ToCharArray();
                int startPtr = 0;
                int endPtr = POLICYNUMBER_LENGTH;
                string policyNumber = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(!inputArray[i].Equals(' '))
                    {
                        policyNumber+= inputArray[i];
                    }
                }
                policyNumbers.Add(policyNumber.ToString());
                startPtr+=POLICYNUMBER_LENGTH;
                endPtr+=POLICYHOLDERID_LENGTH;
                string holderID = "";
                for(int i=startPtr; i<endPtr;i++)
                {
                    if(!inputArray[i].Equals(' '))
                    {
                        holderID+=inputArray[i];
                    }
                }
                holderIds.Add(holderID.ToString());
                startPtr+=POLICYHOLDERID_LENGTH;
                endPtr+=POLICYHOLDERDOB_LENGTH;
                string dob = "";
                int counter = 0;
                for(int i=startPtr; i <endPtr; i++)
                {   
                    counter++;
                    if(counter == 5)
                    {dob+='/';}
                    if(counter==7)
                    {dob+='/';}
                    dob+=inputArray[i];
                }
                DateTime holderDOB = Convert.ToDateTime(dob);
                holderDOBs.Add(holderDOB);
                startPtr+=POLICYHOLDERDOB_LENGTH;
                endPtr+=FATHERSAGEATDEATH_LENGTH;
                counter = 0;
                string fathersInt = "";
                string fathersDec = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    counter++;
                    if(counter >= 4)
                    {
                        fathersDec+=inputArray[i];
                    }
                    else
                    {
                        fathersInt+=inputArray[i];
                    }
                }
                if(!fathersInt.Contains("N"))
                {
                    double fathersAge = Convert.ToDouble(fathersInt);
                    double fathersMonths = Convert.ToDouble(fathersDec) / 12;
                    fathersAge += (double)fathersMonths;
                    fathersAges.Add(fathersAge);
                }
                else{fathersAges.Add(0);}
                startPtr+=FATHERSAGEATDEATH_LENGTH;
                endPtr+=MOTHERSAGEATDEATH_LENGTH;
                counter = 0;
                string mothersInt = "";
                string mothersDec = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    counter++;
                    if(counter >= 4)
                    {
                        mothersDec+=inputArray[i];
                    }
                    else
                    {
                        mothersInt+= inputArray[i];
                    }
                }
                if(!mothersInt.Contains("N"))
                {
                    double mothersAge = Convert.ToDouble(mothersInt);
                    double mothersMonths = Convert.ToDouble(mothersDec) / 12;
                    mothersAge += mothersMonths;
                    mothersAges.Add(mothersAge);
                }
                else{mothersAges.Add(0);}
                startPtr+=MOTHERSAGEATDEATH_LENGTH;
                endPtr+=CIGS_LENGTH;
                string cigs ="";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if (!inputArray[i].Equals(' '))
                    {
                        cigs+=inputArray[i];
                    }                   
                }
                int cigsPerDay = Convert.ToInt32(cigs);
                cigsPerDays.Add(cigsPerDay);
                startPtr+=CIGS_LENGTH;
                endPtr+=SMOKING_LENGTH;
                counter = 0;
                string smokeInt = "";
                string smokeDec = "";
                for(int i=startPtr; i<endPtr;i++)
                {
                    counter++;
                    if(counter >= 4)
                    {
                        smokeDec+= inputArray[i];
                    }
                    else{
                        smokeInt+=inputArray[i];
                    }
                }
                int smokingHistory = Convert.ToInt32(smokeInt) * 12;
                int smokeMonths = Convert.ToInt32(smokeDec);
                smokingHistory += smokeMonths;
                smokingHistories.Add(smokingHistory);
                startPtr += SMOKING_LENGTH;
                endPtr += BLOODPRESSURE_LENGTH;
                string bloodPressure = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(!inputArray[i].Equals(' '))
                    {
                        bloodPressure += inputArray[i];
                    }
                }
                int blood = (int)Convert.ToUInt32(bloodPressure);
                bloodPressures.Add(blood);
                startPtr+=BLOODPRESSURE_LENGTH;
                endPtr+=AVGFAT_LENGTH;
                string fat = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(!inputArray[i].Equals(' '))
                    {
                        fat+=inputArray[i];
                    }
                }
                int avgFatPerDay = (int)Convert.ToUInt32(fat);
                avgFats.Add(avgFatPerDay);
                startPtr+=AVGFAT_LENGTH;
                endPtr+=HEARTDISEASE_LENGTH;
                bool heartDisease = false;
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(inputArray[i].Equals('Y'))
                    {
                        heartDisease = true;
                    }
                }
                heartDiseases.Add(heartDisease);
                startPtr+=HEARTDISEASE_LENGTH;
                endPtr+=CANCER_LENGTH;
                bool cancer = false;
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(inputArray[i].Equals('Y'))
                    {
                        cancer = true;
                    }
                }
                cancers.Add(cancer);
                startPtr+=CANCER_LENGTH;
                endPtr+=HOSPITALIZED_LENGTH;
                bool hospitalized = false;
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(inputArray[i].Equals('Y'))
                    {
                        hospitalized = true;
                    }
                }
                hospitalizeds.Add(hospitalized);
                startPtr+=HOSPITALIZED_LENGTH;
                endPtr+=DANGEROUS_LENGTH;
                string dangerousAct = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(!inputArray[i].Equals(' '))
                    {
                        dangerousAct+=inputArray[i];
                    }
                }
                dangerousActs.Add(dangerousAct.ToString());
                startPtr+=DANGEROUS_LENGTH;
                endPtr+=POLICYSTART_LENGTH;
                string startDate = "";
                counter = 0;
                for(int i=startPtr; i<endPtr;i++)
                {
                    counter++;
                    if(counter==5)
                    {
                        startDate+='/';
                    }
                    if(counter==7)
                    {
                        startDate+='/';
                    }
                    startDate+=inputArray[i];
                }
                DateTime policyStart = Convert.ToDateTime(startDate);
                policyStarts.Add(policyStart);
                startPtr+=POLICYSTART_LENGTH;
                endPtr+=POLICYEND_LENGTH;
                counter = 0;
                string endDate = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    if(counter==5)
                    {
                        endDate+='/';
                    }
                    if(counter==7)
                    {
                        endDate+='/';
                    }
                    endDate+=inputArray[i];
                }
                DateTime policyEndDate;
                if(DateTime.TryParse(endDate,out policyEndDate))
                {
                    policyEnds.Add(policyEndDate);
                }
                startPtr+=POLICYEND_LENGTH;
                endPtr+=AGENTID_LENGTH;
                string agentID = "";
                for(int i=startPtr; i<endPtr; i++)
                {
                    agentID+= inputArray[i];
                }
                agentIds.Add(agentID.ToString());
                startPtr+=AGENTID_LENGTH;
                endPtr+=PAYOFF_LENGTH;
                counter = 0;
                string payoff = "";
                string payoffDec = "";
                for(int i=startPtr; i<endPtr;i++)
                {
                    counter++;
                    if(counter >= 9)
                    {
                        payoffDec+=inputArray[i];
                    }
                    else
                    {
                        payoff += inputArray[i];
                    }
                }
                double payoffAmount = Convert.ToDouble(payoff);
                payoffAmount += Convert.ToDouble(payoffDec) / 100;
                payoffs.Add(payoffAmount);
                startPtr+=PAYOFF_LENGTH;
                endPtr+=PREMIUM_LENGTH;
                string premium = "";
                string premiumDec = "";
                counter = 0;
                for(int i=startPtr; i<endPtr; i++)
                {
                    counter++;
                    if(counter >= 9)
                    {
                        premiumDec+=inputArray[i];
                    }
                    else
                    {
                        premium += inputArray[i];
                    }
                }
                double monthlyPremium = Convert.ToDouble(premium);
                monthlyPremium += Convert.ToDouble(premiumDec)/100;
                premiums.Add(monthlyPremium);                   
            }
        }
    }
}
