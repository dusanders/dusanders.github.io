barberShop.start(); for(int i=0; i<15; i++) { //create a customer Customer customer = new Customer(i, barberShop); customer.start(); //have them enter at different times try { Thread.sleep(150); } catch (InterruptedException ex) {} } } } class Customer extends Thread { private final int id; private final BarberShop barberShop; private boolean cut; public Customer(int inID, BarberShop inBarber) { this.id = inID; this.barberShop = inBarber; this.cut = false; } @Override public void run() { try { //notify there is a customer barberShop.customer.release(); //if there are available chairs if(barberShop.hasChair()) { //while we didnt get a haircut.. while(!cut) { //take a chair barberShop.takeChair(this.id); //wait for the barber barberShop.barber.acquire(); //get our haircut barberShop.cutHair(this.id); //release the barber barberShop.barber.release(); //ok to exit loop cut = true; } } else { //there were no chairs, we leave System.out.println("Customer " + id + " has left!"); //customer left barberShop.customer.acquire(); } } catch (InterruptedException ex) {} } } class BarberShop extends Thread { public Semaphore customer = new Semaphore(0); public Semaphore barber = new Semaphore(1); private int availableChairs; public BarberShop(int chairSize) { availableChairs = chairSize; } @Override public void run() { //infinite loop while(true) { try { //wait for a customer to arrive customer.acquire(); } catch (InterruptedException ex) {} } } //method to return whether are open chairs public boolean hasChair() { return availableChairs > 0; } //method to 'take' a chair public void takeChair(int id) { System.out.println("Customer " + id+ " has taken a chair!"); availableChairs--; System.out.println("Open seats: " + availableChairs); } //cut the customer's hair public void cutHair(int id) { System.out.println("Customer " + id+ " is getting a cut!"); try { //open a seat availableChairs++; Thread.sleep(505); System.out.println("Customer " + id + " is finished!"); }catch (InterruptedException ex){} } }
