/* Author: Dustin Anderson ** Title: Hw3_4 ** Class: CSCI 451 ** Desc: Implement the one-lane bridge problem. */ package onelanebridge; public class OneLaneBridge { public static void main(String[] args) { //create the bridge Bridge bridge = new Bridge(); for(int i=0; i < 10; i++) { //create two sets of trucks new Truck(i, bridge).start(); new Truck2(i+20, bridge).start(); } } } class Bridge { //method to cross the bridge public void traverse(int id) { try { System.out.println("Truck " + id + " started going!"); //travel the bridge Thread.sleep(300); System.out.println("Truck " + id + " finished the bridge!"); } catch (InterruptedException ex) {} } } class Truck extends Thread { //save our id value private int id; //reference to shared object private final Bridge bridge; //constructor expects an id value and reference to shared bridge public Truck(int inID, Bridge inBridge) { id = inID; bridge = inBridge; } //synchronized traverse() call upon run() @Override public void run() { synchronized(bridge) { bridge.traverse(id); } } } class Truck2 extends Thread { private final int id; private final Bridge bridge; public Truck2(int inID, Bridge inBridge) { id = inID; bridge = inBridge; } @Override public void run() { synchronized(bridge) { bridge.traverse(id); } } }
