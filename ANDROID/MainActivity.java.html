128      @Override 
129      public boolean onOptionsItemSelected(MenuItem item) { 
130          // Handle action bar item clicks here. The action bar will 
131          // automatically handle clicks on the Home/Up button, so long 
132          // as you specify a parent activity in AndroidManifest.xml. 
133          int id = item.getItemId(); 
134          //noinspection SimplifiableIfStatement 
135   
136          //exit the app 
137          if (id == R.id.action_settings) { 
138              super.onDestroy(); 
139              wakeLock.release(); 
140              //free up our resources 
141              if(recorder != null){ 
142                  //get rid of Recorder 
143                  recorder.deactivate(); 
144              } 
145              if(dashCamera.isRunning()) { 
146                  //release our camera 
147                  dashCamera.deactivate(); 
148              } 
149              //delete cache 
150              if(!fileManager.getSaveState()) { 
151                  //delete cache videos 
152                  fileManager.cleanAll(); 
153              } 
154              //shutdown threads 
155              cameraThreads.shutdown(); 
156              audioManager.setStreamSolo(AudioManager.STREAM_VOICE_CALL, false); 
157              //GTFO!! 
158              System.exit(0); 
159          } 
160          return super.onOptionsItemSelected(item); 
161      }

163      public static void launch(){ 
164          //make sure we cant 'launch' twice! 
165          if(!launched) { 
166              //add our preview 
167              previewFrame.addView(dashCamera.getCameraPreview()); 
168              //we need a delay before starting the mediaRecorder to allow views to settle 
169              final Handler delay = new Handler(); 
170              //postDelay this action, wait 1000ms (1sec) 
171              delay.postDelayed(new Runnable() { 
172                  @Override 
173                  public void run() { 
174                      //get our surface 
175                      previewSurface = dashCamera.getCameraPreview().getSurface(); 
176                      recorder = new Recorder(dashCamera, mainActivity, previewSurface); 
177                      recorder.activate(); 
178                  } 
179              }, 1000); 
180              //start the recording! 
181              cameraThreads.scheduleAtFixedRate(new Runnable() { 
182                  @Override 
183                  public void run() { 
184                      if (recorder.isRunning()) { 
185                          recorder.deactivate(); 
186                      } 
187                      //start new Recorder 
188                      recorder = new Recorder(dashCamera, mainActivity, previewSurface); 
189                      recorder.activate(); 
190                      //if we are NOT saving files, clean them up 
191                      if (!fileManager.getSaveState()) { 
192                          fileManager.clean(); 
193                      } 
194                  } 
195              }, 2, 2, TimeUnit.MINUTES); 
196              launched = true; 
197          } 
198      }
